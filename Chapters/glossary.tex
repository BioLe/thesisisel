\newglossaryentry{horizontal-scaling}{
    name = {Horizontal scaling},
    description = {In horizontal scaling, the system is scaled by reducing or increasing the number of servers. It does not allow the system to alter the amount of resources a machine has allocated while it is running.}
}

\newglossaryentry{vertical-scaling}{
    name = {Vertical scaling},
    description = {In vertical scaling, a machine's amount of allocated resources (e.g., RAM, CPU) can be altered dynamically, even without rebooting the machine.}
}

\newglossaryentry{codebase}{
    name = {codebase},
    description = {The complete body of source code for a given software program or application.}
}

\newglossaryentry{pull-model}{
    name = {pull model},
    description = {Pull model is a style of network communication where the initial request for data originates from the client, and then is responded to by the server.}
}

\newglossaryentry{load-balancer}{
    name = {load balancer},
    description = {A load balance routes requests to the appropriate server, distributing the load as a way to maximize velocity and capacity in the system \cite{load-balancing}}
}

\newglossaryentry{n-gram}{
    name = {n-gram},
    description = {An n-gram is a contiguous sequence of n items from a given sample of text or speech. These items can be phonemes, syllables, letters, words or base pairs according to the application.}
}

\newglossaryentry{event-streaming}{
    name = {event streaming},
    description = {Event streaming is the practice of capturing data in real-time from event sources like databases, sensors, software applications in the form of streams of events; storing these events durably for later retrieval; manipulating, processing, and reacting to the event streams in real-time or retrospectively; and routing the event streams to different destination technologies as needed. Even streaming thus ensures a continuous flow and interpretation of data so that the information is at the right place, at the right time.}
}
